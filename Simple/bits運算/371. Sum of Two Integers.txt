class Solution {
    public int getSum(int a, int b) {
        if(b==0)
        {
            return a;
        }
        else
        {
            int sum=a^b;// 符號 ^ (值不同給1)
            int carry=(a&b)<<1;//左移 符號& (值同為1給1)
            return getSum(sum,carry);
            
        }
    }
}
